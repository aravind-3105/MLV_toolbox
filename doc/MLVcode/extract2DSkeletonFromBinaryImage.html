0022 function [fluxImage,skeletonImage,distImage,thin_boundary]=extract2DSkeletonFromBinaryImage(binaryImage,threshold)
0023 
0024 thin_boundary = double(~bwmorph(~binaryImage,'skel',inf));
0025 
0026 
0027 number_of_samples = 60;
0028 epsilon = 1 ;
0029 fraction = 0.025;
0030 area_threshold = max(floor(fraction*max(size(binaryImage,1),size(binaryImage,2))),1);
0031 
0032 
0033 % Computing Gradient Vector Field
0034 %fprintf('Distance function and gradient vector field is being computed ...\n');
0035 [distImage,IDX] = compute_gradient_vector_field(thin_boundary);
0036 % Consider a sphere with radius 1 with some sample points on that
0037 sphere_points = sample_sphere_2D(number_of_samples);
0038 % Computing Average outward flux
0039 %fprintf('Average outward flux is being computed ...\n');
0040 fluxImage = compute_aof(distImage,IDX,sphere_points,epsilon);
0041 skeletonImage = fluxImage;
0042 skeletonImage(fluxImage < threshold*number_of_samples) = 0;
0043 skeletonImage(fluxImage > threshold*number_of_samples) = 1;
0044 skeletonImage = bwmorph(skeletonImage,'skel',inf);
0045 
0046 % refine skeleton
0047 skeletonImage= bwareaopen(skeletonImage,area_threshold);
0048 
0049 
0050 end
