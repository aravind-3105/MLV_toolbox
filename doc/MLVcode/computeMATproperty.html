0020 % The default value of K is 5
0021 % We should specify the list
0022 
0023 function [skeletonImageWithRating,skeletalBranches] = computeMATproperty(MAT,property,skeletalBranches,K)
0024 
0025 if nargin < 3
0026     skeletalBranches = traceSkeleton(MAT);
0027 end
0028 if nargin < 4
0029     K = 5;
0030 end
0031 skeletonImageWithRating = zeros(size(MAT.skeleton));
0032 
0033 for i = 1 : length(skeletalBranches)   
0034     scores = computeMATpropertyPerBranch(skeletalBranches(i),property,K);    
0035     skeletalBranches(i).(property) = scores;
0036     curBranchInds = sub2ind(size(MAT.skeleton),skeletalBranches(i).X,skeletalBranches(i).Y);
0037     skeletonImageWithRating(curBranchInds) = scores;
0038 end
