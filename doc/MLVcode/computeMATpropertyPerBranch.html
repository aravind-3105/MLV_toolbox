0020 % here is the main scoring function --> please note that we originally
0021 % started with saliency scores based on symmetry and later added separation
0022 % this does not mean all the scores computed here are just symmetry
0023 
0024 function result = computeMATpropertyPerBranch(curBranch,property,K)
0025 
0026 
0027 
0028 N = length(curBranch.X);
0029 [R,dR,dX,dY]=getBranchDerivative(curBranch);
0030 result = zeros(N,1);
0031 
0032 
0033 
0034 switch lower(property)
0035     case 'parallelism'  
0036         % this is computing the first derivative of the arc length
0037         skeletalAxisLength = (dX.^2+dY.^2).^.5;
0038         arcLengthVar = (dX.^2+dY.^2+dR.^2).^.5;
0039 
0040         if(N>=3)
0041 
0042             for i = 2 : N-1
0043                 % effective K
0044                 eK = min(min(i-1,N-i),K);
0045                 nom = sum(skeletalAxisLength(i-eK:i+eK));
0046                 denom = sum(arcLengthVar(i-eK:i+eK));
0047                 result(i) = nom/denom;
0048             end        
0049         end
0050         
0051         
0052     case 'separation' 
0053         % this is computing the inverse of the radius function
0054         result = 1-1./R;
0055     
0056     case 'taper'
0057         % this is computing the second derivative of the arc length
0058         dR = smoothdata(dR);
0059         ddR = diff(dR);
0060         if(length(ddR) >= 1)
0061             newddR = [ddR;ddR(end)];
0062         else
0063             newddR = dR;
0064         end
0065         ddR = newddR;
0066         ddR = smoothdata(ddR);
0067 
0068         skeletalAxisLength = (dX.^2+dY.^2).^.5;
0069         arcLengthVar = (dX.^2+dY.^2+(ddR).^2).^.5;
0070 
0071         if(N>=3)
0072 
0073             for i = 2 : N-1
0074                 % effective K
0075                 eK = min(min(i-1,N-i),K);
0076                 nom = sum(skeletalAxisLength(i-eK:i+eK));
0077                 denom = sum(arcLengthVar(i-eK:i+eK));
0078                 result(i) = nom/denom;
0079             end        
0080         end
0081         
0082      case 'mirror'   
0083         % computing curvature of the medial axis
0084         X = curBranch.X;
0085         Y = curBranch.Y;
0086         if length(X)> 3
0087             [~,~,result] = breakToLineSegments([X,Y]);
0088         end
0089      
0090     otherwise
0091         
0092         %
0093         error('Unknown property %s',property);
0094        
0095 end
0096 
0097 result = smoothdata(result,'movmean',3);
0098 result = result.^10;
0099 
0100 end

0105 R = branch.Radius;

0117
